% =====================================================================
% Resume style — single source of truth for fonts, colors, spacing,
% section / role / skill formatting, and variant control.
% Designed for ATS (clean text flow, no hidden characters, no two-col)
% AND for human reviewers (navy accent, hairline rules, hierarchy).
% =====================================================================

% --- Font: Source Sans Pro (Adobe). Strong ATS compat (T1 encoding).
\usepackage[default,semibold,scale=0.95]{sourcesanspro}
\usepackage[T1]{fontenc}

\usepackage[margin=0.6in,a4paper]{geometry}
\setcounter{secnumdepth}{0}

% Kill the default first-line indent globally. The header's tagline /
% contact row / metric strip are rendered as separate paragraphs after
% the name, and \parindent was pushing them right relative to the rule
% and the section bodies underneath. With \parindent=0 everything aligns
% flush to the left margin.
\setlength{\parindent}{0pt}

% --- Color palette
\usepackage{xcolor}
\definecolor{accent}{HTML}{1F3A5F}   % deep navy: senior, financial, classic
\definecolor{muted}{HTML}{4A5568}    % readable gray for dates/locations
\definecolor{rule}{HTML}{B5C0CC}     % hairline rule under section / header

% --- Lists
\usepackage{enumitem}
% Itemize tuned for breathable bullet density.
\setlist[itemize]{topsep=0.3em, partopsep=0pt, parsep=0pt, leftmargin=1.2em, itemsep=0.4em}
\setlist[enumerate]{itemsep=0pt, align=left}

% --- Icons (FontAwesome 5). Used in the contact row.
% Glyphs render as font glyphs; ATS parses the adjacent link text fine.
\usepackage{fontawesome5}

% --- Hyperlinks
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    urlcolor=accent,
    linkcolor=accent,
}

% --- Conditional rendering
\usepackage{etoolbox}

% --- Section / subsection format
\usepackage{titlesec}
\titlespacing{\section}{0pt}{1.2em}{0.5em}
\titlespacing{\subsection}{0pt}{0.7em}{0.15em}
\titlespacing{\subsubsection}{0pt}{0.5em}{0.1em}

% Section: navy + bold + uppercase + hairline rule below the title.
% The rule is full \linewidth, drawn 0.5pt thick in a light navy tone.
% NOTE: \MakeUppercase goes in before-code (not format!) so it only
% scopes to the title text. Putting \uppercase in format leaks into
% the after-code and uppercases color names, breaking xcolor.
\titleformat{\section}
  {\color{accent}\Large\bfseries}
  {}{0em}
  {\MakeUppercase}
  [{\vspace{-0.55em}\textcolor{rule}{\rule{\linewidth}{0.5pt}}}]
\titleformat*{\subsection}{\large\bfseries}
\titleformat*{\subsubsection}{\bfseries\scshape}

% --- \role: standardised role header for use inside \subsection{...}
%   \role{Company}{Title}{Dates}{Location}
% All four parts in navy + bold (bold is inherited from \subsection's
% \bfseries; we add \color{accent} and small differentiating shapes):
%   Company  -- bold large navy
%   Title    -- bold large italic navy
%   Date     -- bold normalsize smallcaps navy (right-aligned)
%   Location -- bold normalsize smallcaps navy (right-aligned, after date)
% \texorpdfstring keeps PDF bookmarks clean (hyperref can't bookmark
% \hfill / \scshape / \lowercase tokens cleanly).
\newcommand{\role}[4]{%
  \texorpdfstring{%
    {\color{accent}#1\ \textendash\ \itshape #2}%
    \hfill {\normalsize\color{accent}\bfseries\scshape\oldstylenums{\lowercase{#3}}}%
    \quad {\normalsize\color{accent}\bfseries\scshape\oldstylenums{\lowercase{#4}}}%
  }{#1 \textendash\ #2 (#3 | #4)}%
}

% --- \rside: legacy right-aligned tag (kept for education / fallback)
\newcommand{\rside}[1]{%
  \hfill {\normalfont \color{muted} \scshape\oldstylenums{\lowercase{#1}}}%
}
\newcommand{\aux}[1]{$|$ {\normalfont\textsc{#1}}}

% --- \skillrow: a tag-style skills line.
%   \skillrow{Label}{item · item · item}
% Label in navy bold uppercase, items follow on the same baseline.
\newcommand{\skillrow}[2]{%
  \noindent\textcolor{accent}{\bfseries\uppercase{#1}}\ \ #2\par\vspace{0.3em}%
}

% --- \metricstrip: subtle one-liner of high-signal achievements / scale.
% \footnotesize so the full metric line fits within the text width
% without breaking; tweak font here if you add/remove items.
\newcommand{\metricstrip}[1]{%
  {\footnotesize\color{accent!90}#1\par}%
}

% --- Variant control --------------------------------------------------
% \resumevariant must be defined in the entry-point file (main_full.tex
% or main_min.tex) BEFORE this style file is included.
\providecommand{\resumevariant}{full}

\newcommand{\bb}[1]{\item #1}
\newcommand{\fb}[1]{\ifdefstring{\resumevariant}{full}{\item #1}{}}
\newcommand{\mb}[1]{\ifdefstring{\resumevariant}{min}{\item #1}{}}
\newcommand{\fonly}[1]{\ifdefstring{\resumevariant}{full}{#1}{}}
\newcommand{\monly}[1]{\ifdefstring{\resumevariant}{min}{#1}{}}
